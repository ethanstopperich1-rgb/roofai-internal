# services/roof-lidar/requirements.txt
# Versions are unpinned (or upper-bounded only where strictly necessary)
# so the pdal/pdal:latest image's Python version (currently 3.13)
# resolves to compatible wheels. PDAL itself comes from the base image
# native libs + the `pdal` pip binding installed alongside.

fastapi
uvicorn
pydantic

# Geospatial / point cloud
numpy
scipy
shapely
pyproj
laspy[lazrs]

# DBSCAN for plane segmentation (was Open3D — replaced with sklearn +
# vectorized scipy KDTree because Open3D 0.19 doesn't support Python 3.13
# and the pdal/pdal:latest base image ships 3.13).
scikit-learn

# YOLO object detection (penetrations on ortho render)
ultralytics
torch
torchvision
pillow

# 3DEP S3 access (still imported by older code paths — minor)
boto3

# HTTP
requests

# matplotlib for vectorized point-in-polygon (path.Path.contains_points)
# in isolate_roof — replaces Shapely's O(N) per-point Python loop with
# a single C call. ~30s saved on 100k-point clouds.
matplotlib

# ─── Phase 2 — CGAL / PolyFit ────────────────────────────────────────
# CGAL Python bindings — provides Polygonal_surface_reconstruction_3
# for the watertight roof mesh reconstruction. Requires system
# libcgal-dev + libgmp-dev + libmpfr-dev + libeigen3-dev + libboost-dev,
# installed via apt in modal_app.py's image build.
#
# Marked optional with a try-import in regularize_planes.py so the
# pipeline degrades gracefully to per-plane alpha-shape when CGAL
# isn't available (local dev w/o the system libs, wheel install issues).
#
# CGAL on PyPI ships pre-built wheels for Linux x64 (Modal's runtime)
# but NOT for Windows / arm64 — local dev on Mac/Windows hits the
# fallback path. That's fine; it means the build_facets.py alpha-shape
# pipeline gets exercised in dev and PolyFit gets exercised in prod.
CGAL
