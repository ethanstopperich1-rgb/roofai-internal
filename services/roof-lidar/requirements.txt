# services/roof-lidar/requirements.txt
# Versions are unpinned (or upper-bounded only where strictly necessary)
# so the pdal/pdal:latest image's Python version (currently 3.13)
# resolves to compatible wheels. PDAL itself comes from the base image
# native libs + the `pdal` pip binding installed alongside.

fastapi
uvicorn
pydantic

# Geospatial / point cloud
numpy
scipy
shapely
pyproj
laspy[lazrs]

# DBSCAN for plane segmentation (was Open3D — replaced with sklearn +
# vectorized scipy KDTree because Open3D 0.19 doesn't support Python 3.13
# and the pdal/pdal:latest base image ships 3.13).
scikit-learn

# YOLO object detection (penetrations on ortho render)
ultralytics
torch
torchvision
pillow

# 3DEP S3 access (still imported by older code paths — minor)
boto3

# HTTP
requests

# matplotlib for vectorized point-in-polygon (path.Path.contains_points)
# in isolate_roof — replaces Shapely's O(N) per-point Python loop with
# a single C call. ~30s saved on 100k-point clouds.
matplotlib

# Point2Roof config loader deps — the vendored utils/common_utils.py
# uses easydict for cfg.MODEL.* attribute access and pyyaml to parse
# the model config. Without these, point2roof_wrapper._load_model
# fails at load_stage=imports with ModuleNotFoundError, every call
# silently falls through to frustum-fallback. Caught by the gate
# logs immediately after deploy.
easydict
pyyaml

# ─── Phase 2 — CGAL / PolyFit (DISABLED) ─────────────────────────────
# CGAL Python bindings WERE used for the PolyFit watertight roof mesh
# reconstruction (SECONDARY tier, behind Point2Roof primary).
# Disabled on 2026-05-15: the CGAL pip package has no wheel built for
# Python 3.12 (max supported = 3.11), and the pdal/pdal:latest base
# image we use runs Python 3.12. Pip install fails the entire image
# build trying to resolve a missing wheel.
#
# Impact: zero. PolyFit was a fallback for cases where Point2Roof
# returned low confidence; it itself fell back to alpha-shape if
# CGAL wasn't importable. Removing CGAL just makes that fallback
# happen at module-import time instead of at first-call time —
# same final behavior (alpha-shape path).
#
# To re-enable: either downgrade the image to Python 3.11, or wait
# for a CGAL wheel that supports 3.12 (tracked upstream at
# https://github.com/CGAL/cgal-swig-bindings/issues).
# CGAL  # ← uncomment when 3.12 wheel ships
